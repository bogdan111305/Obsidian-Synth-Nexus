● 1. Вставить задачу с ACL (всё одной командой):

  docker exec -i agg-postgres psql -U postgres -d arenadata << 'EOF'
  DO $$
  DECLARE
    v_ut_id UUID := gen_random_uuid();
    v_acl_id UUID := gen_random_uuid();
  BEGIN
    INSERT INTO utm.ut (id, code, name, status, process_id, trace_id, due_date, context, created_at, updated_at)
    VALUES (v_ut_id, 'TSK-TEST-001', 'Тестовая задача', 'CREATED', 'proc-1', gen_random_uuid(), NOW() + INTERVAL '7 days', '{}', NOW(), NOW());

    INSERT INTO utm.ut_acl (id, ut_id, grant_type, grant_value, perms, updated_at)
    VALUES (v_acl_id, v_ut_id, 'user', 'user-42', ARRAY['READ','WRITE'], NOW());

    RAISE NOTICE 'ut_id = %', v_ut_id;
  END $$;
  EOF

  В выводе будет строка вида NOTICE: ut_id = <uuid> — скопируй этот UUID.

  ---
  2. Подождать ~8 секунд, затем посмотреть документ в OpenSearch:

  docker exec agg-opensearch curl -s "http://localhost:9200/utm-ut-index/_doc/utm:ut:<ВСТАВЬ_UUID>" | python3 -m json.tool

  ---
  Или найти последний документ без UUID:

  docker exec agg-opensearch curl -s "http://localhost:9200/utm-ut-index/_search?size=1&sort=_seq_no:desc" | python3 -m json.tool
