# Перемещение по ревизиям

Перемещение по ревизиям является ключевой функциональностью для работы с аудитом в Hibernate Envers. Это позволяет навигировать по истории изменений сущностей и анализировать их эволюцию во времени.

## Обзор концепции

Ревизии представляют собой версии данных с уникальными номерами. Каждая ревизия содержит снимок состояния сущности на определенный момент времени.

```
Entity: User(id=1)
Revisions: [1, 3, 5, 7, 10]
Timeline: [2023-01-01, 2023-01-15, 2023-02-01, 2023-02-15, 2023-03-01]
```

```java
@Service
public class FilteredRevisionService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public List<Number> getRevisionsByUser(String username) {
        AuditReader auditReader = getAuditReader();
        
        // Получение всех ревизий
        List<Number> allRevisions = auditReader.getRevisions("User", 1L);
        
        // Фильтрация по пользователю
        return allRevisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revision = auditReader.findRevision(CustomRevisionEntity.class, rev);
                return username.equals(revision.getUsername());
            })
            .collect(Collectors.toList());
    }
    
    public List<Number> getRevisionsByDateRange(Date from, Date to) {
        AuditReader auditReader = getAuditReader();
        
        List<Number> allRevisions = auditReader.getRevisions("User", 1L);
        
        return allRevisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revision = auditReader.findRevision(CustomRevisionEntity.class, rev);
                long timestamp = revision.getTimestamp();
                return timestamp >= from.getTime() && timestamp <= to.getTime();
            })
            .collect(Collectors.toList());
    }
}
```

### Навигация по истории

```java
@Service
public class HistoryNavigationService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager);
    }
    
    public List<Object> getEntityHistory(String entityName, Serializable id) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        return revisions.stream()
            .map(rev -> auditReader.find(entityName, id, rev))
            .collect(Collectors.toList());
    }
    
    public List<User> getUserHistory(Long userId) {
        List<Object> history = getEntityHistory("User", userId);
        return history.stream()
            .map(obj -> (User) obj)
            .collect(Collectors.toList());
    }
    
    public Object getLatestVersion(String entityName, Serializable id) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        if (revisions.isEmpty()) {
            return null;
        }
        
        Number latestRevision = revisions.get(revisions.size() - 1);
        return auditReader.find(entityName, id, latestRevision);
    }
    
    public Object getFirstVersion(String entityName, Serializable id) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        if (revisions.isEmpty()) {
            return null;
        }
        
        Number firstRevision = revisions.get(0);
        return auditReader.find(entityName, id, firstRevision);
    }
}
```

```java
@Service
public class DateBasedRevisionService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public Number findRevisionByDate(String entityName, Serializable id, Date targetDate) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        return revisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revision = auditReader.findRevision(CustomRevisionEntity.class, rev);
                Date revisionDate = new Date(revision.getTimestamp());
                return revisionDate.equals(targetDate);
            })
            .findFirst()
            .orElse(null);
    }
    
    public Number findRevisionBeforeDate(String entityName, Serializable id, Date targetDate) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        return revisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revision = auditReader.findRevision(CustomRevisionEntity.class, rev);
                Date revisionDate = new Date(revision.getTimestamp());
                return revisionDate.before(targetDate);
            })
            .max(Number::compareTo)
            .orElse(null);
    }
    
    public Number findRevisionAfterDate(String entityName, Serializable id, Date targetDate) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        return revisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revision = auditReader.findRevision(CustomRevisionEntity.class, rev);
                Date revisionDate = new Date(revision.getTimestamp());
                return revisionDate.after(targetDate);
            })
            .min(Number::compareTo)
            .orElse(null);
    }
}
```

### Поиск по содержимому

```java
@Service
public class ContentBasedRevisionService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager);
    }
    
    public List<Number> findRevisionsByContent(String entityName, Serializable id, 
                                             String fieldName, Object value) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        return revisions.stream()
            .filter(rev -> {
                Object entity = auditReader.find(entityName, id, rev);
                Object fieldValue = getFieldValue(entity, fieldName);
                return Objects.equals(fieldValue, value);
            })
            .collect(Collectors.toList());
    }
    
    private Object getFieldValue(Object entity, String fieldName) {
        try {
            Field field = entity.getClass().getDeclaredField(fieldName);
            field.setAccessible(true);
            return field.get(entity);
        } catch (Exception e) {
            log.error("Ошибка при получении значения поля", e);
            return null;
        }
    }
    
    public List<Number> findRevisionsByUserEmail(Long userId, String email) {
        return findRevisionsByContent("User", userId, "email", email);
    }
    
    public List<Number> findRevisionsByProductName(Long productId, String name) {
        return findRevisionsByContent("Product", productId, "name", name);
    }
}
```

## Пагинация ревизий

### Простая пагинация

```java
@Service
public class PaginatedRevisionService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager);
    }
    
    public Page<Object> getRevisionsPage(String entityName, Serializable id, 
                                        int page, int size) {
        AuditReader auditReader = getAuditReader();
        List<Number> allRevisions = auditReader.getRevisions(entityName, id);
        
        int totalElements = allRevisions.size();
        int totalPages = (int) Math.ceil((double) totalElements / size);
        
        int startIndex = page * size;
        int endIndex = Math.min(startIndex + size, totalElements);
        
        List<Object> content = allRevisions.subList(startIndex, endIndex)
            .stream()
            .map(rev -> auditReader.find(entityName, id, rev))
            .collect(Collectors.toList());
        
        return new PageImpl<>(content, PageRequest.of(page, size), totalElements);
    }
    
    public Page<Number> getRevisionNumbersPage(String entityName, Serializable id, 
                                              int page, int size) {

        List<Number> content = allRevisions.subList(startIndex, endIndex);
```

### Расширенная пагинация с сортировкой

```java
@Service
public class AdvancedPaginatedRevisionService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public Page<CustomRevisionEntity> getRevisionsWithMetadata(String entityName, Serializable id,
                                                              int page, int size, String sortBy) {
        AuditReader auditReader = getAuditReader();
        List<Number> allRevisions = auditReader.getRevisions(entityName, id);
        
        // Получение метаданных ревизий
        List<CustomRevisionEntity> revisionsWithMetadata = allRevisions.stream()
            .map(rev -> auditReader.findRevision(CustomRevisionEntity.class, rev))
            .collect(Collectors.toList());
        
        // Сортировка
        if ("timestamp".equals(sortBy)) {
            revisionsWithMetadata.sort(Comparator.comparing(CustomRevisionEntity::getTimestamp));
        } else if ("username".equals(sortBy)) {
            revisionsWithMetadata.sort(Comparator.comparing(CustomRevisionEntity::getUsername));
        }
        
        // Пагинация
        int totalElements = revisionsWithMetadata.size();
        int totalPages = (int) Math.ceil((double) totalElements / size);
        
        int startIndex = page * size;
        int endIndex = Math.min(startIndex + size, totalElements);
        
        List<CustomRevisionEntity> content = revisionsWithMetadata.subList(startIndex, endIndex);
        
        return new PageImpl<>(content, PageRequest.of(page, size), totalElements);
    }
}
```

## Сравнение ревизий

### Сравнение двух ревизий

```java
@Service
public class RevisionComparisonService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager);
    }
    
    public Map<String, Object> compareRevisions(String entityName, Serializable id, 
                                               Number revision1, Number revision2) {
        AuditReader auditReader = getAuditReader();
        
        Object entity1 = auditReader.find(entityName, id, revision1);
        Object entity2 = auditReader.find(entityName, id, revision2);
        
        return compareEntities(entity1, entity2);
    }
    
    private Map<String, Object> compareEntities(Object entity1, Object entity2) {
        Map<String, Object> differences = new HashMap<>();
        
        if (entity1 == null || entity2 == null) {
            return differences;
        }
        
        Field[] fields = entity1.getClass().getDeclaredFields();
        
        for (Field field : fields) {
            field.setAccessible(true);
            
            try {
                Object value1 = field.get(entity1);
                Object value2 = field.get(entity2);
                
                if (!Objects.equals(value1, value2)) {
                    differences.put(field.getName(), Map.of(
                        "old", value1,
                        "new", value2
                    ));
                }
            } catch (IllegalAccessException e) {
                log.error("Ошибка при сравнении полей", e);
            }
        }
        
        return differences;
    }
    
    public Map<String, Object> compareWithPrevious(String entityName, Serializable id, 
                                                  Number currentRevision) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(entityName, id);
        
        int currentIndex = revisions.indexOf(currentRevision);
        if (currentIndex <= 0) {
            return new HashMap<>();
        }
        
        Number previousRevision = revisions.get(currentIndex - 1);
        return compareRevisions(entityName, id, previousRevision, currentRevision);
    }
}
```

## Заключение

Перемещение по ревизиям предоставляет мощные возможности для анализа истории изменений в Hibernate Envers:

- **Навигация**: Легкое перемещение между версиями сущностей
- **Фильтрация**: Поиск конкретных ревизий по различным критериям
- **Сравнение**: Анализ различий между версиями
- **Пагинация**: Эффективная работа с большими объемами данных
- **Поиск**: Нахождение конкретных версий по содержимому

При работе с навигацией по ревизиям важно помнить о производительности, правильной обработке ошибок и эффективном использовании памяти.
