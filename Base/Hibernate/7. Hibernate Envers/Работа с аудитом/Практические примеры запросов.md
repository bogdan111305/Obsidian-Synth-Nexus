# Практические примеры запросов

Этот раздел содержит реальные примеры запросов к аудиту в Hibernate Envers, которые можно использовать в различных бизнес-сценариях.

## Анализ изменений пользователей

### Отслеживание изменений профиля пользователя

```java
@Service
public class UserChangeAnalysisService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public List<UserChangeEvent> getUserChangeHistory(Long userId) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(User.class, userId);
        
        List<UserChangeEvent> changeEvents = new ArrayList<>();
        
        for (int i = 1; i < revisions.size(); i++) {
            Number currentRevision = revisions.get(i);
            Number previousRevision = revisions.get(i - 1);
            
            User currentUser = auditReader.find(User.class, userId, currentRevision);
            User previousUser = auditReader.find(User.class, userId, previousRevision);
            CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, currentRevision);
            
            UserChangeEvent changeEvent = createChangeEvent(currentUser, previousUser, revisionInfo);
            changeEvents.add(changeEvent);
        }
        
        return changeEvents;
    }
    
    private UserChangeEvent createChangeEvent(User currentUser, User previousUser, CustomRevisionEntity revisionInfo) {
        UserChangeEvent event = new UserChangeEvent();
        event.setRevisionNumber(revisionInfo.getRev());
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        event.setIpAddress(revisionInfo.getIpAddress());
        
        // Определение изменений
        Map<String, FieldChange> fieldChanges = new HashMap<>();
        
        if (!Objects.equals(currentUser.getUsername(), previousUser.getUsername())) {
            fieldChanges.put("username", new FieldChange(previousUser.getUsername(), currentUser.getUsername()));
        }
        
        if (!Objects.equals(currentUser.getEmail(), previousUser.getEmail())) {
            fieldChanges.put("email", new FieldChange(previousUser.getEmail(), currentUser.getEmail()));
        }
        
        if (!Objects.equals(currentUser.getStatus(), previousUser.getStatus())) {
            fieldChanges.put("status", new FieldChange(previousUser.getStatus(), currentUser.getStatus()));
        }
        
        event.setFieldChanges(fieldChanges);
        return event;
    }
    
    public List<UserChangeEvent> getCriticalUserChanges(Long userId) {
        List<UserChangeEvent> allChanges = getUserChangeHistory(userId);
        
        return allChanges.stream()
            .filter(change -> isCriticalChange(change))
            .collect(Collectors.toList());
    }
    
    private boolean isCriticalChange(UserChangeEvent change) {
        // Критические изменения: изменение статуса, email, роли
        return change.getFieldChanges().containsKey("status") ||
               change.getFieldChanges().containsKey("email") ||
               change.getFieldChanges().containsKey("role");
    }
    
    public List<UserChangeEvent> getUserChangesByDateRange(Long userId, Date fromDate, Date toDate) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(User.class, userId);
        
        return revisions.stream()
            .filter(rev -> {
                CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, rev);
                long timestamp = revisionInfo.getTimestamp();
                return timestamp >= fromDate.getTime() && timestamp <= toDate.getTime();
            })
            .map(rev -> {
                User user = auditReader.find(User.class, userId, rev);
                CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, rev);
                return createSimpleChangeEvent(user, revisionInfo);
            })
            .collect(Collectors.toList());
    }
    
    private UserChangeEvent createSimpleChangeEvent(User user, CustomRevisionEntity revisionInfo) {
        UserChangeEvent event = new UserChangeEvent();
        event.setRevisionNumber(revisionInfo.getRev());
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        return event;
    }
}
```

### Анализ изменений транзакций

```java
@Service
public class FinancialTransactionAnalysisService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public List<TransactionChangeEvent> getTransactionChangeHistory(Long transactionId) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(Transaction.class, transactionId);
        
        List<TransactionChangeEvent> changeEvents = new ArrayList<>();
        
        for (int i = 1; i < revisions.size(); i++) {
            Number currentRevision = revisions.get(i);
            Number previousRevision = revisions.get(i - 1);
            
            Transaction currentTransaction = auditReader.find(Transaction.class, transactionId, currentRevision);
            Transaction previousTransaction = auditReader.find(Transaction.class, transactionId, previousRevision);
            CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, currentRevision);
            
            TransactionChangeEvent changeEvent = createTransactionChangeEvent(currentTransaction, previousTransaction, revisionInfo);
            changeEvents.add(changeEvent);
        }
        
        return changeEvents;
    }
    
    private TransactionChangeEvent createTransactionChangeEvent(Transaction current, Transaction previous, 
                                                            CustomRevisionEntity revisionInfo) {
        TransactionChangeEvent event = new TransactionChangeEvent();
        event.setRevisionNumber(revisionInfo.getRev());
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        
        Map<String, FieldChange> fieldChanges = new HashMap<>();
        
        if (!Objects.equals(current.getAmount(), previous.getAmount())) {
            fieldChanges.put("amount", new FieldChange(previous.getAmount(), current.getAmount()));
        }
        
        if (!Objects.equals(current.getStatus(), previous.getStatus())) {
            fieldChanges.put("status", new FieldChange(previous.getStatus(), current.getStatus()));
        }
        
        if (!Objects.equals(current.getDescription(), previous.getDescription())) {
            fieldChanges.put("description", new FieldChange(previous.getDescription(), current.getDescription()));
        }
        
        event.setFieldChanges(fieldChanges);
        return event;
    }
    
    public List<TransactionChangeEvent> getLargeAmountChanges(BigDecimal threshold) {
        AuditReader auditReader = getAuditReader();
        
        // Получение всех транзакций с изменениями суммы
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(Transaction.class, null)
            .addProjection(AuditEntity.id())
            .addProjection(AuditEntity.property("amount"))
            .addProjection(AuditEntity.revisionNumber())
            .add(AuditEntity.property("amount").gt(threshold))
            .getResultList();
        
        List<TransactionChangeEvent> largeChanges = new ArrayList<>();
        
        for (Object[] result : results) {
            Long transactionId = (Long) result[0];
            BigDecimal amount = (BigDecimal) result[1];
            Number revision = (Number) result[2];
            
            TransactionChangeEvent event = createLargeAmountEvent(transactionId, amount, revision);
            largeChanges.add(event);
        }
        
        return largeChanges;
    }
    
    private TransactionChangeEvent createLargeAmountEvent(Long transactionId, BigDecimal amount, Number revision) {
        AuditReader auditReader = getAuditReader();
        CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, revision);
        
        TransactionChangeEvent event = new TransactionChangeEvent();
        event.setRevisionNumber(revision);
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        
        Map<String, FieldChange> fieldChanges = new HashMap<>();
        fieldChanges.put("amount", new FieldChange(null, amount));
        event.setFieldChanges(fieldChanges);
        
        return event;
    }
    
    public List<TransactionChangeEvent> getSuspiciousTransactions() {
        AuditReader auditReader = getAuditReader();
        
        // Поиск подозрительных транзакций (множественные изменения статуса)
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(Transaction.class, null)
            .addProjection(AuditEntity.id())
            .addProjection(AuditEntity.revisionNumber())
            .addProjection(AuditEntity.revisionProperty("username"))
            .add(AuditEntity.property("status").eq("PENDING"))
            .getResultList();
        
        return results.stream()
            .map(result -> {
                Long transactionId = (Long) result[0];
                Number revision = (Number) result[1];
                String username = (String) result[2];
                
                return createSuspiciousTransactionEvent(transactionId, revision, username);
            })
            .collect(Collectors.toList());
    }
    
    private TransactionChangeEvent createSuspiciousTransactionEvent(Long transactionId, Number revision, String username) {
        AuditReader auditReader = getAuditReader();
        CustomRevisionEntity revisionInfo = auditReader.findRevision(CustomRevisionEntity.class, revision);
        
        TransactionChangeEvent event = new TransactionChangeEvent();
        event.setRevisionNumber(revision);
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(username);
        event.setSuspicious(true);
        
        return event;
    }
}
```

### Отслеживание изменений ролей и прав доступа

```java
@Service
public class SecurityAuditService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, SecurityRevisionEntity.class);
    }
    
    public List<SecurityChangeEvent> getSecurityChangeHistory(Long userId) {
        AuditReader auditReader = getAuditReader();
        List<Number> revisions = auditReader.getRevisions(User.class, userId);
        
        List<SecurityChangeEvent> securityEvents = new ArrayList<>();
        
        for (int i = 1; i < revisions.size(); i++) {
            Number currentRevision = revisions.get(i);
            Number previousRevision = revisions.get(i - 1);
            
            User currentUser = auditReader.find(User.class, userId, currentRevision);
            User previousUser = auditReader.find(User.class, userId, previousRevision);
            SecurityRevisionEntity revisionInfo = auditReader.findRevision(SecurityRevisionEntity.class, currentRevision);
            
            SecurityChangeEvent securityEvent = createSecurityChangeEvent(currentUser, previousUser, revisionInfo);
            if (securityEvent.hasSecurityChanges()) {
                securityEvents.add(securityEvent);
            }
        }
        
        return securityEvents;
    }
    
    private SecurityChangeEvent createSecurityChangeEvent(User currentUser, User previousUser, 
                                                       SecurityRevisionEntity revisionInfo) {
        SecurityChangeEvent event = new SecurityChangeEvent();
        event.setRevisionNumber(revisionInfo.getRev());
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        event.setIpAddress(revisionInfo.getIpAddress());
        event.setUserAgent(revisionInfo.getUserAgent());
        event.setRiskScore(revisionInfo.getRiskScore());
        
        Map<String, FieldChange> securityChanges = new HashMap<>();
        
        // Отслеживание изменений роли
        if (!Objects.equals(currentUser.getRole(), previousUser.getRole())) {
            securityChanges.put("role", new FieldChange(previousUser.getRole(), currentUser.getRole()));
            event.setCriticalChange(true);
        }
        
        // Отслеживание изменений статуса аккаунта
        if (!Objects.equals(currentUser.getAccountStatus(), previousUser.getAccountStatus())) {
            securityChanges.put("accountStatus", new FieldChange(previousUser.getAccountStatus(), currentUser.getAccountStatus()));
            event.setCriticalChange(true);
        }
        
        // Отслеживание изменений прав доступа
        if (!Objects.equals(currentUser.getPermissions(), previousUser.getPermissions())) {
            securityChanges.put("permissions", new FieldChange(previousUser.getPermissions(), currentUser.getPermissions()));
            event.setCriticalChange(true);
        }
        
        event.setSecurityChanges(securityChanges);
        return event;
    }
    
    public List<SecurityChangeEvent> getHighRiskSecurityChanges() {
        AuditReader auditReader = getAuditReader();
        
        // Поиск изменений с высоким риском
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(User.class, null)
            .addProjection(AuditEntity.id())
            .addProjection(AuditEntity.revisionNumber())
            .addProjection(AuditEntity.revisionProperty("riskScore"))
            .add(AuditEntity.revisionProperty("riskScore").gt(70))
            .getResultList();
        
        return results.stream()
            .map(result -> {
                Long userId = (Long) result[0];
                Number revision = (Number) result[1];
                Integer riskScore = (Integer) result[2];
                
                return createHighRiskSecurityEvent(userId, revision, riskScore);
            })
            .collect(Collectors.toList());
    }
    
    private SecurityChangeEvent createHighRiskSecurityEvent(Long userId, Number revision, Integer riskScore) {
        AuditReader auditReader = getAuditReader();
        SecurityRevisionEntity revisionInfo = auditReader.findRevision(SecurityRevisionEntity.class, revision);
        
        SecurityChangeEvent event = new SecurityChangeEvent();
        event.setRevisionNumber(revision);
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        event.setRiskScore(riskScore);
        event.setHighRisk(true);
        
        return event;
    }
    
    public List<SecurityChangeEvent> getSuspiciousLoginAttempts() {
        AuditReader auditReader = getAuditReader();
        
        // Поиск подозрительных попыток входа
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(User.class, null)
            .addProjection(AuditEntity.id())
            .addProjection(AuditEntity.revisionNumber())
            .addProjection(AuditEntity.revisionProperty("ipAddress"))
            .add(AuditEntity.revisionProperty("riskScore").gt(50))
            .add(AuditEntity.revisionProperty("username").like("%admin%"))
            .getResultList();
        
        return results.stream()
            .map(result -> {
                Long userId = (Long) result[0];
                Number revision = (Number) result[1];
                String ipAddress = (String) result[2];
                
                return createSuspiciousLoginEvent(userId, revision, ipAddress);
            })
            .collect(Collectors.toList());
    }
    
    private SecurityChangeEvent createSuspiciousLoginEvent(Long userId, Number revision, String ipAddress) {
        AuditReader auditReader = getAuditReader();
        SecurityRevisionEntity revisionInfo = auditReader.findRevision(SecurityRevisionEntity.class, revision);
        
        SecurityChangeEvent event = new SecurityChangeEvent();
        event.setRevisionNumber(revision);
        event.setTimestamp(revisionInfo.getTimestamp());
        event.setModifiedBy(revisionInfo.getUsername());
        event.setIpAddress(ipAddress);
        event.setSuspiciousLogin(true);
        
        return event;
    }
}
```

## Генерация отчетов

### Отчет по изменениям за период

```java
@Service
public class AuditReportService {
    
    @Autowired
    private EntityManager entityManager;
    
    private AuditReader getAuditReader() {
        return AuditReaderFactory.get(entityManager, CustomRevisionEntity.class);
    }
    
    public AuditReport generateAuditReport(Date fromDate, Date toDate) {
        AuditReader auditReader = getAuditReader();
        
        AuditReport report = new AuditReport();
        report.setFromDate(fromDate);
        report.setToDate(toDate);
        report.setGeneratedAt(new Date());
        
        // Статистика по типам сущностей
        Map<String, Long> entityTypeStats = getEntityTypeStatistics(auditReader, fromDate, toDate);
        report.setEntityTypeStatistics(entityTypeStats);
        
        // Статистика по пользователям
        Map<String, Long> userStats = getUserStatistics(auditReader, fromDate, toDate);
        report.setUserStatistics(userStats);
        
        // Критические изменения
        List<CriticalChangeEvent> criticalChanges = getCriticalChanges(auditReader, fromDate, toDate);
        report.setCriticalChanges(criticalChanges);
        
        // Топ изменений
        List<TopChangeEvent> topChanges = getTopChanges(auditReader, fromDate, toDate);
        report.setTopChanges(topChanges);
        
        return report;
    }
    
    private Map<String, Long> getEntityTypeStatistics(AuditReader auditReader, Date fromDate, Date toDate) {
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(null, null)
            .addProjection(AuditEntity.entityClass())
            .addProjection(AuditEntity.count("id"))
            .add(AuditEntity.revisionProperty("timestamp").ge(fromDate.getTime()))
            .add(AuditEntity.revisionProperty("timestamp").le(toDate.getTime()))
            .addGroup(AuditEntity.entityClass())
            .getResultList();
        
        Map<String, Long> stats = new HashMap<>();
        for (Object[] result : results) {
            String entityType = ((Class<?>) result[0]).getSimpleName();
            Long count = (Long) result[1];
            stats.put(entityType, count);
        }
        
        return stats;
    }
    
    private Map<String, Long> getUserStatistics(AuditReader auditReader, Date fromDate, Date toDate) {
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(null, null)
            .addProjection(AuditEntity.revisionProperty("username"))
            .addProjection(AuditEntity.count("id"))
            .add(AuditEntity.revisionProperty("timestamp").ge(fromDate.getTime()))
            .add(AuditEntity.revisionProperty("timestamp").le(toDate.getTime()))
            .addGroup(AuditEntity.revisionProperty("username"))
            .addOrder(AuditEntity.count("id").desc())
            .getResultList();
        
        Map<String, Long> stats = new HashMap<>();
        for (Object[] result : results) {
            String username = (String) result[0];
            Long count = (Long) result[1];
            stats.put(username, count);
        }
        
        return stats;
    }
    
    private List<CriticalChangeEvent> getCriticalChanges(AuditReader auditReader, Date fromDate, Date toDate) {
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(User.class, null)
            .addProjection(AuditEntity.id())
            .addProjection(AuditEntity.revisionNumber())
            .addProjection(AuditEntity.revisionProperty("username"))
            .addProjection(AuditEntity.revisionProperty("timestamp"))
            .add(AuditEntity.revisionProperty("timestamp").ge(fromDate.getTime()))
            .add(AuditEntity.revisionProperty("timestamp").le(toDate.getTime()))
            .add(AuditEntity.property("role").eq("ADMIN"))
            .addOrder(AuditEntity.revisionProperty("timestamp").desc())
            .setMaxResults(10)
            .getResultList();
        
        return results.stream()
            .map(result -> {
                Long userId = (Long) result[0];
                Number revision = (Number) result[1];
                String username = (String) result[2];
                Long timestamp = (Long) result[3];
                
                return new CriticalChangeEvent(userId, revision, username, new Date(timestamp));
            })
            .collect(Collectors.toList());
    }
    
    private List<TopChangeEvent> getTopChanges(AuditReader auditReader, Date fromDate, Date toDate) {
        List<Object[]> results = auditReader.createQuery()
            .forEntitiesModifiedAtRevision(null, null)
            .addProjection(AuditEntity.entityClass())
            .addProjection(AuditEntity.count("id"))
            .add(AuditEntity.revisionProperty("timestamp").ge(fromDate.getTime()))
            .add(AuditEntity.revisionProperty("timestamp").le(toDate.getTime()))
            .addGroup(AuditEntity.entityClass())
            .addOrder(AuditEntity.count("id").desc())
            .setMaxResults(5)
            .getResultList();
        
        return results.stream()
            .map(result -> {
                String entityType = ((Class<?>) result[0]).getSimpleName();
                Long count = (Long) result[1];
                
                return new TopChangeEvent(entityType, count);
            })
            .collect(Collectors.toList());
    }
}
```

## Заключение

Практические примеры запросов к аудиту в Hibernate Envers предоставляют мощные возможности для:

- **Анализа изменений**: Отслеживание истории изменений сущностей
- **Безопасности**: Мониторинг критических изменений и подозрительной активности
- **Финансового аудита**: Отслеживание изменений транзакций и крупных операций
- **Отчетности**: Генерация комплексных отчетов по аудиту
- **Бизнес-аналитики**: Анализ паттернов изменений и пользовательской активности

При работе с практическими запросами важно помнить о производительности, правильной обработке больших объемов данных и эффективном использовании ресурсов.
